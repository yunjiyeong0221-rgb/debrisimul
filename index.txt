<script>
    // API 없이 작동하기 위한 가상 로직 (템플릿 엔진 방식)
    function generateDebriefing() {
        const scenario = document.getElementById('scenario').value;
        const model = document.getElementById('model-select').value;
        if (!scenario) return alert("시나리오를 입력해 주세요.");

        showLoading(true);

        // 1.5초 후 결과 출력 (AI가 분석하는 느낌 유도)
        setTimeout(() => {
            const result = `
[${model} 모델 적용] 
### 핵심 질문
1. 당시 상황에서 가장 먼저 떠오른 생각은 무엇이었나요?
2. 시나리오 과정 중 목표 달성을 방해한 요소는 무엇이었습니까?
3. 유사한 상황이 재발한다면 어떤 행동 변화를 시도하시겠습니까?

### 스크립트 예시
**교수자:** "시나리오에서 나타난 '${scenario.substring(0, 20)}...' 상황에 대해 이야기해 봅시다. 이 과정에서 본인의 판단 근거는 무엇이었나요?"
<br>**학습자:** (성찰 유도)
            `;
            displayResult(result);
            showLoading(false);
        }, 1500);
    }

    function getStrategicAdvice() {
        const scenario = document.getElementById('scenario').value;
        if (!scenario) return alert("시나리오를 입력해 주세요.");

        showLoading(true);

        setTimeout(() => {
            const result = `
### 교육학적 전략 조언
1. **심리적 안전감 확보**: 학습자가 실수를 두려워하지 않도록 비판적 태도보다는 탐구적 질문을 던지세요.
2. **폐쇄형 질문 지양**: "예/아니오" 보다는 "어떻게", "왜"를 활용한 개방형 질문으로 사고를 확장하세요.
3. **인지적 부하 관리**: 복잡한 상황이므로 핵심적인 성찰 포인트 1~2개에 집중하는 것이 좋습니다.
            `;
            displayResult(result);
            showLoading(false);
        }, 1500);
    }

    function showLoading(isLoading) {
        document.getElementById('loader').style.display = isLoading ? 'block' : 'none';
        if(isLoading) document.getElementById('result-area').style.display = 'none';
    }

    function displayResult(text) {
        let formattedText = text.replace(/### (.*)/g, '<h3 style="color:var(--navy);">$1</h3>')
                                .replace(/\*\*(.*)\*\*/g, '<strong>$1</strong>')
                                .replace(/\[(.*?)\]/g, '<span class="model-badge">$1</span>')
                                .replace(/\n/g, '<br>');
        document.getElementById('result-content').innerHTML = formattedText;
        document.getElementById('result-area').style.display = 'block';
    }
